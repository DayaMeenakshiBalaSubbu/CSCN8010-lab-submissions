#7. Compare the models using the training and validation data. Construct a table summarizing the train validation results. Each model should have a separate row in the table (3 points):
#    1. R-Squared
#    2. Mean Absolute Error (MAE)
#    3. MAPE

# Function to calculate Mean Absolute Percentage Error (MAPE)
def mean_absolute_percentage_error(y_true, y_pred):
    return np.mean(np.abs((y_true - y_pred) / y_true)) * 100

# Dictionary to store output
output = []

# Evaluate each polynomial model
for degree, (p_model, polynomial) in m.items():
    #Transforming the data using the polynomial features
    train_X_polynomial = polynomial.transform(train_X_bmi)
    Validation_X_polynomial = polynomial.transform(Validation_X_bmi)
    # Make predictions and evaluating the metrics and storing them in a list
    train_pred = p_model.predict(train_X_polynomial)
    Validation_pred = p_model.predict(Validation_X_polynomial)
    train_r2 = r2_score(train_y, train_pred)
    Validation_r2 = r2_score(Validation_y, Validation_pred)
    train_mae = mean_absolute_error(train_y, train_pred)
    Validation_mae = mean_absolute_error(Validation_y, Validation_pred)
    train_mape = mean_absolute_percentage_error(train_y, train_pred)
    Validation_mape = mean_absolute_percentage_error(Validation_y, Validation_pred)
    output.append([degree, train_r2, Validation_r2, train_mae, Validation_mae, train_mape, Validation_mape])
